---
title: 'Get All Users'
api: 'GET /admin/user/list'
---

Retrieves a list of all users in the system. This endpoint is restricted to admin users.

## Authentication

Requires admin authentication via Bearer token.

<CodeGroup>
```bash curl
curl -X GET https://api.falconalert.com/admin/user/list \
  -H "Authorization: Bearer YOUR_ADMIN_TOKEN"
```
</CodeGroup>

## Response

<ResponseField name="users" type="array">
  Array of user objects
  
  <Expandable title="User Object">
    <ResponseField name="id" type="number">
      Unique identifier for the user
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Full name of the user
    </ResponseField>
    
    <ResponseField name="email" type="string">
      User's email address
    </ResponseField>
    
    <ResponseField name="username" type="string">
      User's unique username
    </ResponseField>
    
    <ResponseField name="password" type="string">
      Hashed password (should not be exposed to clients)
    </ResponseField>
    
    <ResponseField name="salt" type="string">
      Password salt for hashing
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of user creation
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      ISO 8601 timestamp of last update
    </ResponseField>
    
    <ResponseField name="image_path" type="string">
      Path to user's profile picture
    </ResponseField>
    
    <ResponseField name="role_id" type="number">
      User's role identifier (1 = admin, 2 = user)
    </ResponseField>
  </Expandable>
</ResponseField>
