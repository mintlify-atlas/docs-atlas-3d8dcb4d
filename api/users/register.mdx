---
title: "Register User"
api: "POST /users/register"
description: "Register a new user in the FalconAlert system"
---

## Authentication

This endpoint requires a valid JWT Bearer token.

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
  
  Format: `Bearer <your_access_token>`
</ParamField>

## Request Body

<ParamField body="email" type="string" required>
  User's email address
  
  Example: `name@domain.com`
</ParamField>

<ParamField body="name" type="string" required>
  User's full name
  
  Example: `John Doe`
</ParamField>

<ParamField body="password" type="string" required>
  User's password (minimum 8 characters)
  
  Example: `password123`
</ParamField>

<ParamField body="role_id" type="string" default="1">
  User's role ID
  
  - `1` - Regular user (default)
  - `2` - Admin user
  
  Example: `1`
</ParamField>

## Response

<ResponseField name="user" type="object">
  User information
  
  <ResponseField name="id" type="number">
    Unique user identifier
  </ResponseField>
  
  <ResponseField name="name" type="string">
    User's full name
  </ResponseField>
  
  <ResponseField name="email" type="string">
    User's email address
  </ResponseField>
  
  <ResponseField name="username" type="string">
    Auto-generated username
  </ResponseField>
  
  <ResponseField name="salt" type="string">
    Password salt for security
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    Account creation timestamp (ISO 8601)
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    Last update timestamp (ISO 8601)
  </ResponseField>
  
  <ResponseField name="image_path" type="string">
    Path to user's profile image
  </ResponseField>
  
  <ResponseField name="role_id" type="number">
    User's role identifier
  </ResponseField>
</ResponseField>

<ResponseField name="settings" type="array">
  User settings configuration
  
  <ResponseField name="id" type="number">
    Settings record ID
  </ResponseField>
  
  <ResponseField name="user_id" type="number">
    Associated user ID
  </ResponseField>
  
  <ResponseField name="is_reactions_enabled" type="number">
    Whether reactions are enabled (0 or 1)
  </ResponseField>
  
  <ResponseField name="is_review_enabled" type="number">
    Whether reviews are enabled (0 or 1)
  </ResponseField>
  
  <ResponseField name="is_reports_enabled" type="number">
    Whether reports are enabled (0 or 1)
  </ResponseField>
</ResponseField>

    'Authorization': 'Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...',
