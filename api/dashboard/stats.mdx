---
title: Get Dashboard Stats
api: GET /dashboard
description: Retrieve comprehensive dashboard statistics including total reports, status breakdowns, top categories, monthly trends, top users, recent alerts, and popular reports
---

## Response

<ResponseField name="stats" type="object">
  Overall statistics summary
  
  <Expandable title="stats properties">
    <ResponseField name="total_reports" type="number">
      Total number of reports in the system
    </ResponseField>
    
    <ResponseField name="pending_reports" type="number">
      Number of reports with pending status
    </ResponseField>
    
    <ResponseField name="approved_reports" type="number">
      Number of reports that have been approved
    </ResponseField>
    
    <ResponseField name="rejected_reports" type="number">
      Number of reports that have been rejected
    </ResponseField>
    
    <ResponseField name="protected_people" type="number">
      Number of people protected by approved reports
    </ResponseField>
    
    <ResponseField name="total_users" type="number">
      Total number of users in the system
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topCategoriesReports" type="array">
  List of top categories by report count
  
  <Expandable title="topCategoriesReports item">
    <ResponseField name="name" type="string">
      Category name
    </ResponseField>
    
    <ResponseField name="total_reportes" type="number">
      Total number of reports in this category
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="reportsPerMonth" type="array">
  Monthly report statistics
  
  <Expandable title="reportsPerMonth item">
    <ResponseField name="month" type="string">
      Month in YYYY-MM format
    </ResponseField>
    
    <ResponseField name="total_reports" type="number">
      Total number of reports created in this month
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topUsers" type="array">
  Top users by report creation count
  
  <Expandable title="topUsers item">
    <ResponseField name="name" type="string">
      User's name
    </ResponseField>
    
    <ResponseField name="total_reports" type="number">
      Total number of reports created by this user
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="recentAlerts" type="array">
  Most recently created alerts
  
  <Expandable title="recentAlerts item">
    <ResponseField name="id" type="number">
      Alert ID
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Alert title
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topReportsMonth" type="array">
  Top reports for the current month ranked by upvotes
  
  <Expandable title="topReportsMonth item">
    <ResponseField name="id" type="number">
      Report ID
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Report title
    </ResponseField>
    
    <ResponseField name="upvotes" type="number">
      Number of upvotes received
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl --request GET \
  --url https://api.falconalert.com/dashboard
```

</RequestExample>
